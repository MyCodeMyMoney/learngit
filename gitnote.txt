1. mkdir <filename>  //创建文件夹
2. pwd               //显示当前目录
3. git init          //把当前目录初始化为git仓库
4. git add           //添加文件
5. git commit -m <message>  //提交文件并添加备注
6. git status        //查看仓库状态
7. git diff          //查看修改的内容
8. git log           //查看提交日志
9. git reflog        //查看所有命令
10. git reset --hard HEAD^      //版本回退到上一次的提交
11. git checkout -- <filename>  //版本回退到暂存区的状态(一键还原功能)
12. touch <FileName>            //新建文件
13. reset                       //清屏命令
14. git rm <FileName>           //将文件从版本库中删除
